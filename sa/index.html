// Безопасный конвертер: десятичная → произвольная база
std::string convertFromDecimal(unsigned long long n, int base) {
    if (base < 2 || base > 16)
        throw std::overflow_error("base out of range [2..16]");

    if (n == 0) return "0";
    const std::string digits = "0123456789ABCDEF";
    std::string out;
    while (n > 0) {
        out += digits[n % base];
        n /= base;
    }
    std::reverse(out.begin(), out.end());
    return out;
}

// Валидация: только символы из текущего алфавита
bool isValidNumber(const std::string& s, int base) {
    for (char c : s) {
        int v = std::isdigit(c) ? c-'0' : std::toupper(c)-'A'+10;
        if (v < 0 || v >= base) return false;
    }
    return !s.empty();
}
